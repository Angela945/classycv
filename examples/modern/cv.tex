\begin{document}

% Choose US business letter and modern resume format.
\classycvSet{/settings/cover letter/format}{us-block}
\classycvSet{/settings/resume/format}{modern}

% Set heading options
\classycvSet{/put heading/options/padding}{0pt}
\classycvSet{/put heading/options/skip above}{\baselineskip}
\classycvSet{/put heading/options/skip below}{\baselineskip}

% Set resume entry options.
\classycvSet{/put resume entry/options/skip above}{0pt}
\classycvSet{/put resume entry/options/skip below}{\baselineskip}
